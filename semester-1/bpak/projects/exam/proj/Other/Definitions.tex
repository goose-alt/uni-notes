\newdimen\zerolinewidth

\tikzset{
  zero line width/.code={%
    \zerolinewidth=\pgflinewidth
    \tikzset{line width=0cm}%
  },
  use line width/.code={%
    \tikzset{line width=\the\zerolinewidth}%
  },
  draw with anchor in boundary/.style={
    zero line width,
    postaction={draw,use line width},
  },
}
